% ======== Tikz museful macros==================================================
% ------------------------------------------------------------------------------
% #1 half length of tick
% #2 min x
% #3 min y
% #4 max x
% #5 max y
\newcommand{\myAxes}[5][3pt]{
  %axis
  \draw[-{Stealth[scale=1.5]},evaluate={\x=#4+1;}] (#2,0)--(\x,0);
  \draw[-{Stealth[scale=1.5]},evaluate={\y=#5+1;}] (0,#3)--(0,\y);
  %ticks
  \foreach \x in {#2,...,#4} \draw (\x,#1)--(\x,-#1);
  \foreach \y in {#3,...,#5} \draw (#1,\y)--(-#1,\y);
}
% ------------------------------------------------------------------------------
% #1 --- list of points
% #2 --- list of segments
% #3 --- first point
% #4 --- draw color
% #5 --- fill color
\newcommand{\directedPolygon}[5]{
  \fill[#4] (#3) \foreach \pa/\pb in #2 {-- (\pb)};
  \nodesDef{#1}
  \foreach \pa/\pb in #2 {
    \draw[thick,#5,->,>=stealth] (n\pa) -- (n\pb);
  };
}
% ------------------------------------------------------------------------------
% Usage \arcThroughThoPoints[options]{center}{p}{q};
\newcommand{\arcThroughThoPoints}[4][]{
\coordinate (center) at (#2);
\draw[#1]
 let \p1=($(#3)-(center)$),
     \p2=($(#4)-(center)$),
     \n0={veclen(\p1)},       % Radius
     \n1={atan2(\x1,\y1)}, % angles
     \n2={atan2(\x2,\y2)},
     \n3={\n2>\n1?\n2:\n2+360}
   in (#3) arc(\n1:\n3:\n0);
}
